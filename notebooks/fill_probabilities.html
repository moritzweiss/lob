# Define the time array
t = np.linspace(0, 10, 100)

# Define the exponential damping function
plt.figure(figsize=(10, 6))
plt.title('Exponential Damping Function')
plt.xlabel('Time')
plt.ylabel('Amplitude')
plt.xticks(np.arange(min(t), max(t)+1, 1.0))  # Set x ticks at every integer

scalings = [0.0, 0.1, 0.25, 0.5, 1]

for c in scalings:

    y = np.exp(-c*t)
    # Plot the function
    plt.plot(t, y)

plt.grid(True)
plt.legend([f'c={c}' for c in scalings])
plt.show()
